      sql ALTER TABLE type auto_increment = 1; 3. 对于设置了自增的列, 不传入该列的值,但要传入 null 占位,否则后面列的数据会传到前面的列 4. 增删改都可以用 runner.update(); 方法,它不需要将修改的内容用一个 handler 接收


    • 自创方法–解决删除元素之后,新增加的元素 id 断层问题–在每次删除元素之后设置主键从最大数开始自增

    • 目前一条 runner 的方法执行一条语句,要执行多条可以将 sql 语句分开多次执行

    ALTER TABLE type auto_increment = 1

    思路是给每一个修改按钮绑定事件，并且由该修改按钮出发，相对于该按钮定位该行需要提取的内容

    // 定位每个class=mr-2的元素，为每个元素执行方法
$('.mr-2').each(function (index, element) {
  // 为定位到的元素设置点击事件
  $(this).click(function () {
    /* 相对于该按钮
        挑选其含有class=text-right的父标签
        然后挑选该标签含有class=userId/userName/userTel的同级标签
        分别赋给变量

    let changeUserId = $(this)
      .parents('.text-right')
      .prevAll('.userId')
      .text();
    let changeUserName = $(this)
      .parents('.text-right')
      .prevAll('.userName')
      .text();

      .parents('.text-right')
      .prevAll('.userTel')
      .text();

    $('#changeUserIdDis').attr('value', changeUserId);
    $('#changeUserIdHid').attr('value', changeUserId);

  });
});

<%-- 计算归还时间  --%>
function getReturnDate() {
    // 获取当前时间
    let date = new Date();
    // 转换为毫秒数
    let milliseconds = date.getMilliseconds();
    // 计算归还毫秒数(30天)
    let returnMills = milliseconds += 30 * 24 * 60 * 60 * 1000;
    // 转换为日期
    date.setMilliseconds(returnMills);
    // 转换为字符串
    let year = date.getFullYear();
    // 0~11
    let month = date.getMonth() + 1;
    let day = date.getDate();

    let returnDate = year + "/" + month + "/" + day;
    return returnDate;
}
